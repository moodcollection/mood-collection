async function fetchCollection() { const grid = document.getElementById('main-product-grid'); try { // 1. Fetch the list of products from GitHub const response = await fetch('https://api.github.com/repos/moodcollection/mood-collection/contents/products'); const files = await response.json(); grid.innerHTML = ''; // 2. Loop through each product file for (let i = 0; i < files.length; i++) { const file = files[i]; if (file.name.endsWith('.md')) { const res = await fetch(file.download_url); const text = await res.text(); // 3. Extract Title, Price, and Image URL const title = text.match(/title:\s*"(.*?)"/)?.[1] || "MØØD Item"; const price = text.match(/price:\s*"(.*?)"/)?.[1] || "0.00"; let img = text.match(/image:\s*"(.*?)"/)?.[1] || ""; // IMAGE PATH FIXER: // If Admin says "/images/photo.jpg", we change it to "./images/photo.jpg" if (img.startsWith('/')) { img = '.' + img; } const slug = file.name.replace('.md', ''); // 4. THE 1-2-3-1 LAYOUT LOGIC // Index 0 = Large, 1-2 = Medium, Others = Small let sizeClass = 'small'; if (i === 0) sizeClass = 'large'; else if (i === 1 || i === 2) sizeClass = 'medium'; grid.innerHTML += `
${title}
£${price}
 `; } } } catch (err) { console.error("Error loading products:", err); grid.innerHTML = "

Loading your collection...

"; } } fetchCollection();
